/** * INTERBANDOS 2026 - Backend (Google Apps Script) * ------------------------------------------------ * Lee las hojas "Puntos_General", "Puntos_Curso", "Capitanes" y "Cronograma" * de este Google Sheet y las devuelve como JSON. * * Se llama vía GET con: * ?action=puntos -> { general, curso, totalRojo, totalAzul } * ?action=capitanes -> { rojo: [...], azul: [...] } * ?action=cronograma -> [ {fecha, titulo, descripcion}, ... ] * ?action=todo -> { puntos, capitanes, cronograma } (usa esto la app) * * Siempre requiere &token=DRL194749 (mismo token que usás en tus otros sistemas DRL). */ const TOKEN_VALIDO = "DRL194749"; function doGet(e) { try { const token = e.parameter.token; if (token !== TOKEN_VALIDO) { return respuestaJSON({ error: "Token inválido" }); } const accion = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); if (accion === "puntos") return respuestaJSON(obtenerPuntos(ss)); if (accion === "capitanes") return respuestaJSON(obtenerCapitanes(ss)); if (accion === "cronograma") return respuestaJSON(obtenerCronograma(ss)); if (accion === "deportes") return respuestaJSON(obtenerPuntosDeportes(ss)); if (accion === "todo") { return respuestaJSON({ puntos: obtenerPuntos(ss), capitanes: obtenerCapitanes(ss), cronograma: obtenerCronograma(ss), deportes: obtenerPuntosDeportes(ss) }); } return respuestaJSON({ error: "Acción no reconocida: " + accion }); } catch (err) { return respuestaJSON({ error: err.toString() }); } } function respuestaJSON(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // Convierte una hoja completa (encabezados en fila 1) en un array de objetos. // Los encabezados se normalizan a minúsculas para que coincidan con las claves que usa el frontend. function hojaAJSON(hoja) { if (!hoja) return []; const datos = hoja.getDataRange().getValues(); if (datos.length < 2) return []; const headers = datos[0].map(h => h.toString().trim().toLowerCase()); return datos.slice(1) .filter(fila => fila.join("").toString().trim() !== "") .map(fila => { const obj = {}; headers.forEach((h, i) => obj[h] = fila[i]); return obj; }); } function obtenerPuntos(ss) { const general = hojaAJSON(ss.getSheetByName("Puntos_General")); // columnas: instancia | rojo | azul const curso = hojaAJSON(ss.getSheetByName("Puntos_Curso")); // columnas: curso | rojo | azul let totalRojo = 0, totalAzul = 0; general.forEach(f => { totalRojo += Number(f.rojo) || 0; totalAzul += Number(f.azul) || 0; }); return { general, curso, totalRojo, totalAzul }; } function obtenerCapitanes(ss) { // columnas: banda o bando (Rojo/Azul) | nombre | curso | rol (General/Referente/Capitán/etc) const filas = hojaAJSON(ss.getSheetByName("Capitanes")); const valorBanda = f => (f.banda ?? f.bando ?? "").toString().trim().toLowerCase(); const rojo = filas.filter(f => valorBanda(f) === "rojo"); const azul = filas.filter(f => valorBanda(f) === "azul"); return { rojo, azul }; } function obtenerCronograma(ss) { // columnas: fecha | titulo | descripcion return hojaAJSON(ss.getSheetByName("Cronograma")); } // Hoja "Puntos_Deportes": deporte | curso | categoria | goles_rojo | goles_azul // Cada fila es UN partido. El puntaje (15 gana / 10 y 10 empate / 5 pierde) lo calcula el frontend, // esta funcion solo devuelve los partidos tal cual estan cargados. function obtenerPuntosDeportes(ss) { return hojaAJSON(ss.getSheetByName("Puntos_Deportes"));
